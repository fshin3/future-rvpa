## 将来予測の図
# 親魚資源量と漁獲量の時系列の図示
(g5_future <- plot_futures(res.pma, #vpaの結果

                   future.name=c("F current",str_c("HCR(beta=",future.default$input$HCR$beta,")")),

                   Bban=derive_RP_value(refs.base,"Bban0")$SSB,
